Standard JDK Solution Pattern:

public class BankAccount {
    private double balance;

    public BankAccount(double initialBalance) {
        this.balance = initialBalance > 0 ? initialBalance : 0;
    }

    public double getBalance() {
        return this.balance;
    }

    public void deposit(double amount) {
        if (amount > 0) this.balance += amount;
    }

    public boolean withdraw(double amount) {
        if (amount > 0 && amount <= this.balance) {
            this.balance -= amount;
            return true;
        }
        return false;
    }

    public static void main(String[] args) {
        BankAccount acc = new BankAccount(1000);
        System.out.print("Initial: " + acc.getBalance());
        acc.deposit(500);
        System.out.print(", Post-Deposit: " + acc.getBalance());
        boolean success = acc.withdraw(2000);
        System.out.println(", Rejected: " + (success ? "Accepted" : "Out of funds"));

public class DivisionSafety {
    public static void divide(int num, int den) {
        try {
            int result = num / den;

Standard JDK Solution Pattern:

import java.io.File;
import java.io.FileNotFoundException;
import java.util.Scanner;

public class InventoryReader {
    public static void readInventory(String filePath) {
        File dataFile = new File(filePath);
        if (!dataFile.exists()) {
            System.out.println("Error: File not found on disk.");
            return;
        }

        try (Scanner sc = new Scanner(dataFile)) {
            int lineNum = 1;
            while (sc.hasNextLine()) {
                String line = sc.nextLine();
                String[] tokens = line.split(",");
                if (tokens.length >= 2) {
                    System.out.printf("Row %d: Product: %s | Cost: $%s\n", lineNum++, tokens[0].trim(), tokens[1].trim());
                }
            }
        } catch (FileNotFoundException e) {
            System.out.println("Fatal: " + e.getMessage());
        }
    }

    public static void main(String[] args) {
        // Mock data representation matching expected output
        readInventory("mock_inventory.csv");
    }
}

import javax.swing.JButton;
import javax.swing.JFrame;
import javax.swing.JTextField;
import java.awt.FlowLayout;

public class ClickCounter {
    public static void main(String[] args) {
        JFrame frame = new JFrame("Interactive Counter");
        frame.setLayout(new FlowLayout());

        JTextField display = new JTextField("0", 10);
        display.setEditable(false);
        JButton btn = new JButton("Increment");

        btn.addActionListener(e -> {
            try {
                int count = Integer.parseInt(display.getText());
                display.setText(String.valueOf(count + 1));
            } catch (NumberFormatException ex) {
                display.setText("0");
            }
        });

        frame.add(display);
        frame.add(btn);
        frame.setSize(300, 150);
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.setLocationRelativeTo(null);

public class MenuItem {

    private double price;
    private int    qty;

        this.price = price;

    public double getPrice() { return price; }

import javax.swing.*;
import java.awt.*;
import java.io.*;

// Custom exception
class InvalidScoreException extends Exception {
    public InvalidScoreException(String msg) { super(msg); }
}

// Contract interface
interface Gradable {
    String getGrade();
}

// Encapsulated Class
class Student implements Gradable {
    private String name;
    private int score;

    public Student(String name, int score) throws InvalidScoreException {
        if (score < 0 || score > 100) {
            throw new InvalidScoreException("Score must be between 0 and 100.");
        }
        this.name = name;
        this.score = score;
    }

    public String getName() { return name; }
    public int getScore() { return score; }

    @Override
    public String getGrade() {
        if (score >= 90) return "A";
        if (score >= 80) return "B";
        if (score >= 70) return "C";
        return "F";
    }

    @Override
    public String toString() {
        return name + "," + score + "," + getGrade();
    }
}

public class GradeBookApp {
    private static final String FILE_PATH = "grades.txt";

    public static void main(String[] args) {
        JFrame frame = new JFrame("Grade Book Manager");
        frame.setLayout(new GridBagLayout());
        GridBagConstraints gbc = new GridBagConstraints();
