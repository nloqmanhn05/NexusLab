Standard JDK Solution Pattern:

public class SmartLight {
    private boolean isOn = false;
    private int brightness = 0;

    public void toggle() {
        this.isOn = !this.isOn;
    }

    public void setBrightness(int level) {
        if (level < 0) this.brightness = 0;
        else if (level > 100) this.brightness = 100;
        else this.brightness = level;
    }

    public String getStatus() {
        return "Status: Light is " + (isOn ? "ON" : "OFF") + " at brightness level " + brightness + "%";
    }

    public static void main(String[] args) {
        SmartLight light = new SmartLight();
        light.toggle();
        light.setBrightness(75);
        System.out.println(light.getStatus());
    }
}

Easy OOP-02-01

Encapsulated BankAccount (Accessors, Mutators & Constructors)

Protect account records from external manipulation. Hide variables with private parameters. Expose state strictly via valid deposit() and withdraw() transactions.

Expected Console Output 
Initial: 1000.0, Post-Deposit: 1500.0, Rejected: Out of funds

Secure Withdraw Constraint:

public boolean withdraw(double amount) {
    if (amount > 0 && amount <= balance) {
        this.balance -= amount;
        return true;
    }
    return false;
}

Standard JDK Solution Pattern:

public class BankAccount {
    private double balance;

    public BankAccount(double initialBalance) {
        this.balance = initialBalance > 0 ? initialBalance : 0;
    }

    public double getBalance() {
        return this.balance;
    }

    public void deposit(double amount) {
        if (amount > 0) this.balance += amount;
    }

    public boolean withdraw(double amount) {
        if (amount > 0 && amount <= this.balance) {
            this.balance -= amount;
            return true;
        }
        return false;
    }

    public static void main(String[] args) {
        BankAccount acc = new BankAccount(1000);
        System.out.print("Initial: " + acc.getBalance());
        acc.deposit(500);
        System.out.print(", Post-Deposit: " + acc.getBalance());
        boolean success = acc.withdraw(2000);
        System.out.println(", Rejected: " + (success ? "Accepted" : "Out of funds"));
    }
}

Standard JDK Solution Pattern:

public class DivisionSafety {
    public static void divide(int num, int den) {
        try {
            int result = num / den;
            System.out.println("Result: " + result);
        } catch (ArithmeticException e) {
            System.out.println("Captured Error: Division by zero is prohibited.");
        } finally {
            System.out.println("Finally Block Executed.");
        }
    }

    public static void main(String[] args) {
        divide(10, 0);
    }
}

Standard JDK Solution Pattern:

import java.io.File;
import java.io.FileNotFoundException;
import java.util.Scanner;

public class InventoryReader {
    public static void readInventory(String filePath) {
        File dataFile = new File(filePath);
        if (!dataFile.exists()) {
            System.out.println("Error: File not found on disk.");
            return;
        }

        try (Scanner sc = new Scanner(dataFile)) {
            int lineNum = 1;
            while (sc.hasNextLine()) {
                String line = sc.nextLine();
                String[] tokens = line.split(",");
                if (tokens.length >= 2) {
                    System.out.printf("Row %d: Product: %s | Cost: $%s\n", lineNum++, tokens[0].trim(), tokens[1].trim());
                }
            }
        } catch (FileNotFoundException e) {
            System.out.println("Fatal: " + e.getMessage());
        }
    }

    public static void main(String[] args) {
        // Mock data representation matching expected output
        readInventory("mock_inventory.csv");
    }
}

Standard JDK Solution Pattern:

import javax.swing.JButton;
import javax.swing.JFrame;
import javax.swing.JTextField;
import java.awt.FlowLayout;

public class ClickCounter {
    public static void main(String[] args) {
        JFrame frame = new JFrame("Interactive Counter");
        frame.setLayout(new FlowLayout());

        JTextField display = new JTextField("0", 10);
        display.setEditable(false);
        JButton btn = new JButton("Increment");

        btn.addActionListener(e -> {
            try {
                int count = Integer.parseInt(display.getText());
                display.setText(String.valueOf(count + 1));
            } catch (NumberFormatException ex) {
                display.setText("0");
            }
        });

        frame.add(display);
        frame.add(btn);
        frame.setSize(300, 150);
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.setLocationRelativeTo(null);
        frame.setVisible(true);

Interactive Grade Book Manager

System Architecture Requirements:

  • Inheritance & ADTs: Create a Gradable Interface requiring an evaluation grade.
  • Custom Exception: If the inputted score is below 0 or exceeds 100, raise a custom exception.
  • File Persistence (I/O): Append accepted records to grades.txt, and reload past entries.
  • Swing Graphical Interface: Build a panel utilizing JFrame, JTextField inputs (Name, ID, Score), a save button, and a visual list area.

Implementation Roadmap:

// 2. Define the Gradable interface

// 3. Build Student class with fields + getter/setters

Full Core Project Class:

import javax.swing.*;
import java.awt.*;
import java.io.*;

// Custom exception
class InvalidScoreException extends Exception {
    public InvalidScoreException(String msg) { super(msg); }
}

// Contract interface
interface Gradable {
    String getGrade();
}

// Encapsulated Class
class Student implements Gradable {
    private String name;
    private int score;

    public Student(String name, int score) throws InvalidScoreException {
        if (score < 0 || score > 100) {
            throw new InvalidScoreException("Score must be between 0 and 100.");
        }
        this.name = name;
        this.score = score;
    }

    public String getName() { return name; }
    public int getScore() { return score; }

    @Override
    public String getGrade() {
        if (score >= 90) return "A";
        if (score >= 80) return "B";
        if (score >= 70) return "C";
        return "F";
    }

    @Override
    public String toString() {
        return name + "," + score + "," + getGrade();
    }
}

// GUI Driver class
public class GradeBookApp {
    private static final String FILE_PATH = "grades.txt";

    public static void main(String[] args) {
        JFrame frame = new JFrame("Grade Book Manager");
        frame.setLayout(new GridBagLayout());
        GridBagConstraints gbc = new GridBagConstraints();
        gbc.insets = new Insets(5, 5, 5, 5);
        gbc.fill = GridBagConstraints.HORIZONTAL;

        // UI Components
        JLabel nameLbl = new JLabel("Student Name:");
        JTextField nameTxt = new JTextField(15);
        JLabel scoreLbl = new JLabel("Score (0-100):");
        JTextField scoreTxt = new JTextField(15);
        JButton saveBtn = new JButton("Save Record");
        JTextArea displayArea = new JTextArea(8, 25);
        displayArea.setEditable(false);

        // Layout Assembly
        gbc.gridx = 0; gbc.gridy = 0; frame.add(nameLbl, gbc);
        gbc.gridx = 1; frame.add(nameTxt, gbc);
        gbc.gridx = 0; gbc.gridy = 1; frame.add(scoreLbl, gbc);
        gbc.gridx = 1; frame.add(scoreTxt, gbc);
        gbc.gridx = 0; gbc.gridy = 2; gbc.gridwidth = 2; frame.add(saveBtn, gbc);
        gbc.gridx = 0; gbc.gridy = 3; frame.add(new JScrollPane(displayArea), gbc);

        // Event handler
        saveBtn.addActionListener(e -> {
            try {
                String name = nameTxt.getText().trim();
                int score = Integer.parseInt(scoreTxt.getText().trim());

                if (name.isEmpty()) {
                    JOptionPane.showMessageDialog(frame, "Name cannot be empty!", "Error", JOptionPane.ERROR_MESSAGE);
                    return;
                }

                // Initialize class, test parameters, throw Exception if failed
                Student student = new Student(name, score);

                // Write to file
                try (BufferedWriter writer = new BufferedWriter(new FileWriter(FILE_PATH, true))) {
                    writer.write(student.toString());
                    writer.newLine();
                }

                displayArea.append("Saved: " + student.getName() + " (" + student.getGrade() + ")\n");
                nameTxt.setText("");
                scoreTxt.setText("");

            } catch (NumberFormatException ex) {
                JOptionPane.showMessageDialog(frame, "Please enter an integer!", "Format Error", JOptionPane.ERROR_MESSAGE);
            } catch (InvalidScoreException ex) {
                JOptionPane.showMessageDialog(frame, ex.getMessage(), "Input Range Error", JOptionPane.ERROR_MESSAGE);
            } catch (IOException ex) {
                JOptionPane.showMessageDialog(frame, "File Error: " + ex.getMessage(), "Disk Error", JOptionPane.ERROR_MESSAGE);
            }
        });

        frame.setSize(350, 450);
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.setLocationRelativeTo(null);
        frame.setVisible(true);
    }
}
